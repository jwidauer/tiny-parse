96 const auto result = parse_it(sv);

361template <class T>
362class LessThan : public BaseParser<LessThan<T>> {
363 public:
364 LessThan(size_t max, const T& parser) noexcept : max_{max}, parser_{parser} {}
365 [[nodiscard]] size_t min_length() const noexcept override { return 0; }
366
367 protected:
368 [[nodiscard]] Result parse_it(const std::string_view& sv) const override {
369 auto result = sv >> parser_;
370 auto success = result.success;
371 // Start at 2 because we already ran the parser once and want to stop at
372 // max_ - 1
373 for (size_t i = 2; result.success && i < max_; ++i) {
374 result = result >> parser_;
375 success |= result.success;
