110 const auto result = parse_it(sv);

282template <class T>
283class TINY_PARSE_PUBLIC Times : public BaseParser<Times<T>> {
284 public:
285 Times(size_t times, const T& parser) noexcept : times_{times}, parser_{parser} {}
286
287 constexpr size_t min_length() const noexcept override { return parser_.min_length() * times_; }
288
289 protected:
290 constexpr Result parse_it(const std::string_view& sv) const noexcept override {
291 size_t i = 1;
292 auto result = sv >> parser_;
293 for (; result.success && i < times_; ++i) {

375template <class T>
376class TINY_PARSE_PUBLIC LessThan : public BaseParser<LessThan<T>> {
377 public:
378 LessThan(size_t max, const T& parser) noexcept : max_{max}, parser_{parser} {}
379 constexpr size_t min_length() const noexcept override { return 0; }
380
381 protected:
382 constexpr Result parse_it(const std::string_view& sv) const noexcept override {
383 auto result = sv >> parser_;
384 auto success = result.success;
385 // Start at 2 because we already ran the parser once and want to stop at
386 // max_ - 1
387 for (size_t i = 2; result.success && i < max_; ++i) {
388 result = result >> parser_;
389 success |= result.success;
390 }
391
392 return {result.value, success};
393 }
